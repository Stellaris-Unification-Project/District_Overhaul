decision_generator_privatisation = {
	owned_planets_only = yes
	sound = event_administrative_work
	icon = decision_resources

	resources = {
		category = decisions
		cost = {
		}			
	}
	
	potential = {
		owner = { 
			is_regular_empire = yes
			or = {
				is_ai = yes
				and = {
					is_ai = no
					not = { has_policy_flag = alp_private_businesses_prohibited }
					not = { has_policy_flag = alp_private_businesses_restricted }
					or = {
						has_authority = auth_corporate
						has_ethic = ethic_pacifist
						has_ethic = ethic_fanatic_pacifist
						has_ethic = ethic_egalitarian
						has_ethic = ethic_fanatic_egalitarian
					}
				}
			}
		}
		not = { has_planet_flag = pf_privatise_generators }
		num_districts = { type = district_generator value > 0 }
		not = { has_planet_flag = pf_filter_vanilla }
		### DistrictOverhaul
		has_district_generator = yes
		### Check AlphaMod enabled
		has_global_flag = gf_alphamod_activated
	}
	
	effect = {
		set_planet_flag = pf_privatise_generators
	}
	
	ai_weight = {
		weight = 0
		modifier = {
			weight = 50
			or = {
				num_districts = { type = district_generator value > 1 }
				owner = { 
					has_monthly_income = {
						resource = sr_wealth
						value < 25
					}			
				}
			}
		}
	}
}

		}
	}
	
	potential = {
		not = { has_planet_flag = pf_filter_vanilla }
		habitable_structure = no
		not = { has_planet_flag = auto_generators }
		owner = { is_machine_empire = no }
		NOT = { is_planet_class = pc_machine }
		NOT = { is_planet_class = pc_hive }
		NOT = { is_planet_class = pc_city }	
		### DistrictOverhaul
		has_district_generator = yes
		### Check AlphaMod enabled
		has_global_flag = gf_alphamod_activated
	}
	
	allow = {
		num_districts = { type = district_generator value > 0 }
	}
	
	effect = {
		custom_tooltip = decision_auto_generators_custom
		set_planet_flag = auto_generators
	}
	
	ai_weight = {
		weight = 0
		modifier = {
			weight = 5
			num_pops > 25
			num_pops < 80
			count_owned_pops = { 
				limit = { 
					OR = { 
						is_unemployed = yes 
						has_job = servant 
					}
				}
				count < 1
			}			
		}
	}
}
